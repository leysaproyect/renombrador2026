        const downloadBtn = document.getElementById('downloadBtn');

        const error = document.getElementById('error');

        const errorMsg = document.getElementById('errorMsg');

 

        let currentFile = null;

        let extractedCURP = null;

 

        // Eventos de arrastrar y soltar

        uploadArea.addEventListener('click', () => fileInput.click());

            if (files.length > 0) {

                handleFile(files[0]);

            }

        });

 

        fileInput.addEventListener('change', (e) => {

            if (e.target.files.length > 0) {

                handleFile(e.target.files[0]);

            }

        });

        async function handleFile(file) {

            resetUI();

            

            if (file.type !== 'application/pdf') {

                showError('Por favor selecciona un archivo PDF válido.');

                return;

            }

 

            currentFile = file;

            filename.textContent = file.name;

            fileInfo.classList.add('show');

            loading.classList.add('show');

 

            try {

                const curp = await extractCURPFromPDF(file);

                

                if (curp) {

                    extractedCURP = curp;

                    curpValue.textContent = curp;

                    

                    // Generar nuevo nombre de archivo

                    const extension = file.name.split('.').pop();

                    const newName = `${curp}.${extension}`;

                    newFilename.textContent = `Nuevo nombre: ${newName}`;

                    

                    curpDisplay.classList.add('show');

                    downloadBtn.classList.add('show');

                    loading.classList.remove('show');

                } else {

                    showError('No se encontró un CURP válido en el PDF. Asegúrate de que el documento contenga un CURP mexicano válido (18 caracteres).');

                }

            } catch (err) {

                showError('Error al leer el PDF: ' + err.message);

            }

        }

 

        async function extractCURPFromPDF(file) {

            const arrayBuffer = await file.arrayBuffer();

            const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;

            

            let fullText = '';

            

            // Leer todas las páginas

            for (let i = 1; i <= pdf.numPages; i++) {

                const page = await pdf.getPage(i);

                const textContent = await page.getTextContent();

                const pageText = textContent.items.map(item => item.str).join(' ');

                fullText += pageText + ' ';

            }

 

            // Patrón regex para CURP mexicano (18 caracteres)

            // Formato: 4 letras + 6 números + 2 letras + 3 números + 1 letra/número

            const curpPattern = /[A-Z]{4}[0-9]{6}[A-Z]{2}[0-9]{3}[A-Z0-9]/g;

            

            // Buscar todas las coincidencias

            const matches = fullText.match(curpPattern);

            

            if (matches && matches.length > 0) {

                // Filtrar CURPs válidos (18 caracteres exactos)

                const validCURPs = matches.filter(curp => curp.length === 18);

                

                if (validCURPs.length > 0) {

                    // Retornar el primer CURP válido encontrado

                    return validCURPs[0];

                }

            }

 

            // Búsqueda alternativa: buscar "CURP:" o "CURP" seguido del valor

            const altPattern = /CURP[:\\s]+([A-Z0-9]{18})/i;

            const altMatch = fullText.match(altPattern);

            if (altMatch) {

                return altMatch[1];

            }

 

            return null;

        }

 

        downloadBtn.addEventListener('click', () => {

            if (!currentFile || !extractedCURP) return;

 

            const extension = currentFile.name.split('.').pop();

            const newName = `${extractedCURP}.${extension}`;

 

            // Crear enlace de descarga

            const url = URL.createObjectURL(currentFile);

            const a = document.createElement('a');

            a.href = url;

            a.download = newName;

            document.body.appendChild(a);

            a.click();

            document.body.removeChild(a);

            URL.revokeObjectURL(url);

 

            // Feedback visual

            downloadBtn.textContent = '✅ Descargado!';

            downloadBtn.style.background = 'linear-gradient(135deg, #4caf50 0%, #45a049 100%)';

            

            setTimeout(() => {

                downloadBtn.textContent = '⬇️ Descargar archivo renombrado';

                downloadBtn.style.background = 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)';

            }, 2000);

        });
